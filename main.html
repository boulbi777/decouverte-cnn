def build_CNN(input_dim, output_dim, lr=0.001):
    
    # Initialisation du rseau
    model  = Sequential()
    
    #Diffrentes couches
    model.add(Conv2D(filters=30, kernel_size=(3,3), input_shape=input_dim))
    model.add(Activation(activation="relu"))
    model.add(MaxPooling2D(pool_size=(2, 2)))
    model.add(Dropout(rate=0.1))

    #-----------------------------------------------------------------------------------#

    model.add(Conv2D(filters=60, kernel_size=(3,3)))
    model.add(Activation(activation="relu"))
    model.add(MaxPooling2D(pool_size=(2, 2)))
    model.add(Dropout(rate=0.1))

    #----------------------------------------------------------------------------------#

    model.add(Conv2D(filters=120, kernel_size=(3,3)))
    model.add(Activation(activation="relu"))
    model.add(MaxPooling2D(pool_size=(2, 2)))
    model.add(Dropout(rate=0.1))

    #----------------------------------------------------------------------------------#

    model.add(Flatten())
    model.add(Dense(units=150))
    model.add(Activation(activation="relu"))
    model.add(Dropout(rate=0.3))

    #----------------------------------------------------------------------------------#

    model.add(Dense(units=output_dim))
    model.add(Activation(activation="softmax"))    
    
    # initialiser l'optimiseur RMSprop
    opt = rmsprop(learning_rate=lr, decay=1e-6)
    
    #Compilation du modle
    model.compile(loss='categorical_crossentropy', optimizer=opt, metrics=['accuracy'])
    
    return model

3. A quoi sert le DropOut ?¶

Dans la dfinition de la fonction, plusieurs couches sont dfinies les unes aprs les autres. Sur chaque couche, on prcise le type de convultion souhait (2D dans notre cas), la fonction d'activation, etc...

Le droupOut dans un rseau de neurone sert essentiellement viter de loverfitting lors de l'entranement du rseau. En effet, dans les CNN, le passage plusieurs fois sur les mmes pixels d'une image donne (principe de la convolution) peut rapidement conduire ce phnomne de surapprentissge o les neurones des couches caches (ou intermdiaire) ajustent leurs poids qui se collent trop aux donnes d'entranement. Le dropOut est donn avec un pourcentage qui dtermine la proportion de neurones tuer la sortie de chaque couche cache o elle est spcifie (c'est le paramtre 'rate' dans la fonction), et ceci, afin d'viter une surconcentration d'informations. Tuer un neurone reviend simplement fixer son poids 0. Ainsi, le dropOut a pour objectif de permettre une bonne capacit de gnralisation du rseau de neurones construit afin d'amliorer sa capacit de prdiction au vu de nouvelles images. Cette a dj fait ses preuves dans le monde du deep learning. Les proportions de neurones tuer varient gnralement entre 20% et 50%.

La Data Augmentation est un procd qui permet de modifier ou d'accrotre les exemples d'entrainement de notre CNN afin que le rseau de neurones puisse explorer d'autres exemples et augmenter sa capacit de gnralisation. Elle consiste apporter des modifications des images (zoom, rotation, etc...) afin de multiplier les images d'entrainement et de validation. Cette technique donne souvent de bons rsultats pour combattre le surapprentissage. Nous allons toujours utiliser la classe ImageDataGenerator de keras en lui apportant simplement des modifications avec les paramtres de Data Augmentation que nous allons mettre en place. Il est extrmement important de noter que la mthode de Data Augmentation implmente dans keras ne consiste pas multiplier les images d'entrainement mais applique chaque image des modifications alatoires choisies en laissant constant le nombre d'images dans les bases d'entrainement et validation (/!\).

def drop_last(model):
    name_last_layer = str(model.layers[-1])

    model2 = Sequential()
    for layer in model.layers:
        if str(layer) != name_last_layer:
            model2.add(layer)
    return model2

def transfert_learnig_CNN(pretrained_model, nb_last_layers_to_drop, output_dim):
    
    #couper autant de fois la dernire couche demande
    for _ in range(nb_last_layers_to_drop):
        pretrained_model = drop_last(pretrained_model)
    
    #Empilement des modles
    x = Dense(units=50)(pretrained_model.get_output_at(-1))
    x = Dense(units=output_dim)(x)
    o = Activation(activation="softmax")(x)
    
    opt = rmsprop(learning_rate=0.0001, decay=1e-6)
    model2 = Model(inputs=pretrained_model.input, outputs=[o])
    
    model2.compile(loss='categorical_crossentropy', optimizer=opt, metrics=['accuracy'])
    return model2
